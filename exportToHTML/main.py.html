import time
import pywhatkit
import keyboard
import PySimpleGUI as sg

sg.theme('Blue')

layout = [

          [sg.Column([[sg.Image(filename='icon.png')]],justification='center')],
          [sg.Text('Mesajınızı Girin'), sg.Text(size=(15,1))],
          [sg.InputText()],
          [sg.Button('Tamam')]]

window = sg.Window("WhatSalk", icon='icon.ico').Layout(layout)

while True:  # Event Loop
    event, values = window.read()
    print(event, values)
    if event == sg.WIN_CLOSED or event == 'Tamam':
        break
message = values[0]
list = ["+905323918154"]
listleng = str(len(list))
listlengint = len(list)
print("Number of list items: " + listleng + " different numbers are found.")
print("message " + message + " ||")
i = 0
while (i < listlengint):
    a = list[i]
    print(a)
    pywhatkit.sendwhatmsg_instantly(a, message, 5)
    keyboard.press_and_release('enter')
    time.sleep(3)
    keyboard.press_and_release('ctrl+w')
    i = i + 1
